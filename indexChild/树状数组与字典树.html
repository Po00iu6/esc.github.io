#include <iostream>
#include <algorithm>
using namespace std;
#define N 1000001 //数组最大值
int n,m,tot,rt[N],ans;
struct node{
    int ls[N<<5],rs[N<<5],v[N<<5];//小一点就RE
    void cp(int x,int y){
        ls[x]=ls[y],rs[x]=rs[y],v[x]=v[y];//节点复制
    }
    void fix(int pl,int pr,int l,int r,int x){
        cp(pr,pl),v[pr]+=x;//按照pl新建一个节点pr(两棵树可以共享节点)
        if(l==r)return;
        int mid=(l+r)>1;
        if(mid>=x) fix(ls[pl],ls[pr]=++tot,l,mid,x);
        else fix(rs[pl],rs[pr]=++tot,mid+1,r,x);
    }
    int sum(int pl,int pr,int l,int r,int x,int y){//得出pr和pl之间的大于x小于y的数的和
        if(v[pr]-v[pl]==0) return 0;
        if(l>=x&&r<=y) return v[pr]-v[pl];
        int ret=0,mid=(l+r)>1;
        if(mid>=x) ret+=sum(ls[pl],ls[pr],l,mid,x,y);
        if(mid<y) ret+=sum(rs[pl],rs[pr],mid+1,r,x,y);
        return ret;
    }
}tree;
int main(){
    cin>n;
    for(int i=1;i<=n;i++){
        int x;
        cin>x;
        tree.fix(rt[i-1],rt[i]=++tot,1,N<<10,x);
    }
    cin>m;
    for(int i=1;i<=m;i++){
        int L,R,tmp,lst=0,now=0;

            if(tmp){
                lst=now+1;now+=tmp;
            }
            else break;
        }
        cout<<now+1<<'\n';
    }
    return 0;
}
P3834 【模板】可持久化线段树 2提高+/省选-
#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
#define N 200010
int n,q,m,cnt;// n为序列长度，q为查询数量，m为去重后的元素数量，cnt为节点计数器
int a[N],b[N],tree[N];// a为原始序列，b为排序后的唯一元素序列，tree为可持久化线段树的根节点数组
int sum[N<<5],L[N<<5],R[N<<5];// sum为线段树节点存储的区间和（此处为元素个数），L和R为左右子节点指针
// 构建空线段树
inline int build(int l,int r){
    int mid=(l+r)>>1;
    int rt=++cnt; // 新建节点
    sum[rt]=0;// 初始化区间和为0
    if(l<r){
        L[rt]=build(l,mid);// 递归构建左子树
        R[rt]=build(mid+1,r);// 递归构建右子树
    }
    return rt;// 返回当前节点编号 
}
// 更新线段树，插入新元素
inline int update(int pre,int l,int r,int x){
    int mid=(l+r)>>1;
    int rt=++cnt;// 新建节点
    L[rt]=L[pre];R[rt]=R[pre];sum[rt]=sum[pre]+1;// 复制前驱节点的信息，并增加区间和
    if(l<r){
        if(x<=mid) L[rt]=update(L[pre],l,mid,x);// 如果x在左子树范围内，更新左子树
        else R[rt]=update(R[pre],mid+1,r,x);// 否则更新右子树
    }
    return rt;// 返回当前节点编号
}
// 查询区间第k小值
inline int query(int u,int v,int l,int r,int k){
    int mid=(l+r)>>1;
    if(l>=r)return l;// 如果区间长度为1，直接返回该位置的值（但此处不会执行，因为k一定小于区间长度）
    int x=sum[L[v]]-sum[L[u]];// 计算v和u在左子树中的区间和之差，即[l, mid]区间内v有而u没有的元素个数
    if(x>=k)return query(L[u],L[v],l,mid,k);// 如果左子树中的元素个数大于等于k，继续在左子树中查询
    else return query(R[u],R[v],mid+1,r,k-x);// 否则，在右子树中查询第(k-x)小的元素  
}
int main()
{
    cin>>n>>q;// 输入序列长度和查询数量  
    for(int i=1;i<=n;i++){
        cin>>a[i];
        b[i]=a[i];// 将a数组的元素复制到b数组，用于后续排序和去重  
    }
    sort(b+1,b+n+1); // 对b数组进行排序  
    m=unique(b+1,b+n+1)-b-1;// 去重，并计算去重后的元素数量  
    // 构建可持久化线段树  
    tree[0]=build(1,m);// 构建空线段树作为根节点（虽然实际上没有用到，但为了统一接口）  
    for(int i=1;i<=n;i++){
        int t=lower_bound(b+1,b+m+1,a[i])-b;// 在b数组中找到a[i]的位置（排序后的） 
        tree[i]=update(tree[i-1],1,m,t);// 基于前一个节点的线段树，插入新元素a[i]  
    }
    // 查询并输出结果
    while(q--){
        int x,y,z;
        cin>>x>>y>>z;// 输入查询的区间[x, y]和k值 
        int t=query(tree[x-1],tree[y],1,m,z); // 在线段树中查询区间[x, y]内的第z小值  
        cout<<b[t]<<'\n';// 输出查询结果  
    }
    return 0;
}
*这段代码实现了可持久化权值线段树，用于解决静态区间第k小值的问题。通过构建一系列版本的线段树，每个版本都基于前一个版本进行更新，从而可以在O(log n)的时间内查询任意区间的第k小值。

P5283 [十二省联考 2019] 异或粽子区间[l,r]的异或和等于sr⊕sl-1因为每个si都异或了两遍而抵消。
#include <iostream>
#include <queue>
#include <cstring>
using namespace std;
#define ll long long
#define maxn 501000
int n,k;
ll a[maxn];
int trie[maxn*32][2],sz[maxn*32],tot=0;
void insert(ll val){
    int u=0;
    for(int i=31;i>=0;i--){
        int z=(val>>i)&1;
        if(!trie[u][z]){
            trie[u][z]=++tot;
        }
        u=trie[u][z];
        sz[u]++;
    }
}
ll find(ll val,int t){
    ll s=0;
    int u=0;
    for(int i=31,z;i>=0;i--){
        z=(val>>i)&1;
        if(sz[trie[u][z^1]]>=t){
            u=trie[u][z^1];
            s|=(1ll<<i);
        }else{
            t-=sz[trie[u][z^1]];
            u=trie[u][z];
        }
    }
    return s;
}
struct node{
    ll w;
    int x,y;//x的第y大
    bool operator<(const node &A)const{
        return w<A.w;
    }
};
node d[maxn];
priority_queue<node> q;
int main(){
    cin>>n>>k;
    for(int i=1;i<=n;i++){
        cin>>a[i];
        a[i]^=a[i-1];
    }
    for(int i=0;i<=n;i++){
        insert(a[i]);
    }
    k=2*k;
    for(int i=0;i<=n;i++){
        d[i].w=find(a[i],1);
        d[i].x=i;
        d[i].y=1;
        q.push(d[i]);
    }
    ll ans=0;
    node A;
    for(int i=1;i<=k;i++){
        A=q.top();
        ans+=A.w;
        q.pop();
        d[A.x].y++;
        d[A.x].w=find(a[A.x],d[A.x].y);
        q.push(d[A.x]);
    }
    ans/=2;
    cout<<ans;
    return 0;
}

#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
struct node{
    int v,w,nxt;
}edge[2000001];
int head[2000001];
int cnt=1;
void add(int u,int v,int w){
    edge[++cnt].nxt=head[u];
    edge[cnt].v=v;
    edge[cnt].w=w;
    head[u]=cnt;
}
int sum[2000001];
void dfs(int x,int fa){//预处理 
    for(int i=head[x];~i;i=edge[i].nxt){
        int v=edge[i].v;
        int w=edge[i].w;
        if(v!=fa){
            sum[v]=sum[x]^w;
            dfs(v,x);
        }
    }
}
struct trie{
    int ch[2];
}t[2000001];
int tot;
void build(int val,int x){
    for(int i=(1<<30);i;i>>=1){
        bool c=val&i;//取出二进制下这个数的当前位置 
        if(!t[x].ch[c]){
            t[x].ch[c]=++tot;
        }
        x=t[x].ch[c];
    }
}
int query(int val,int x){
    int ans=0;
    for(int i=(1<<30);i;i>>=1){
        bool c=val&i;
        if(t[x].ch[!c]){//如果这一位可以进行异或就沿着这一条往下走 
            ans+=i;
            x=t[x].ch[!c];
        }
        else x=t[x].ch[c];//否则就沿着另一条路往下走 
    }
    return ans;
}
int main()
{
    memset(head,-1,sizeof(head));
    int n;
    cin>>n;
    for(int i=1;i<=n-1;i++){
        int u,v,w;
        cin>>u>>v>>w;
        add(u,v,w);
        add(v,u,w);
    }
    dfs(1,-1);//预处理出每一个节点到根的异或和 
    for(int i=1;i<=n;i++){//建立trie树 
        build(sum[i],0);
    }
    int ans=0;
    for(int i=1;i<=n;i++){
        ans=max(ans,query(sum[i],0));//查询，取最大值 
    }
    cout<<ans;
}

P1481 魔族密码Trie字典树题目。
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
char c[90];
int word[1000],g[1000][1000];
int ans,res,u,tot,n;
void f(){
    int len=strlen(c);
    u=res=0;
    for(int i=0;i<len;i++){
        int a=c[i]-'0';
        if(!g[u][a])g[u][a]=++tot;
        u=g[u][a];
        res+=word[u];
    }
    word[u]++;
    ans=max(ans,res+1);
}
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++){
        memset(c,0,sizeof(0));
        cin>>c;
        f();
    }
    cout<<ans;
    return 0;
}

#include <iostream>
using namespace std;
#define maxn 500010
int n,m;
int a[maxn],c[maxn],b[maxn];
int lowbit(int x){
    return x&(-x);
}
void add(int x,int y){
    for(int i=x;i<=n;i+=lowbit(i))
        c[i]+=y;
}
int sum(int x){
    int res=0;
    for(int i=x;i;i-=lowbit(i))
        res+=c[i];
    return res;
}

    }

#include <iostream>

#define int long long

struct node{
    int val;
}tree[4000001];
int a[1000100];
void build(int root,int l,int r){
    if(l==r){
        tree[root].val=a[l];
        return;
    }
    int m=(l+r)>>1;
    build(root<<1,l,m);
    build(root<<1|1,m+1,r);
    tree[root].val=tree[root<<1].val+tree[root<<1|1].val;
}
void update(int root,int l,int r,int ql,int qr,int k){
    if(qr<l||r<ql)return;//这一句必须加，不加就卡住了 
    if(ql<=l&&r<=qr){
        tree[root].val+=k;
        return;
    }
    int m=(l+r)>>1;
    update(root<<1,l,m,ql,qr,k);
    update(root<<1|1,m+1,r,ql,qr,k);
    tree[root].val=tree[root<<1].val+tree[root<<1|1].val;
}
int query(int root,int l,int r,int ql,int qr){
    if(qr<l||r<ql)return 0;//这一句必须加，不加就卡住了 
    if(ql<=l&&r<=qr){
        return tree[root].val;
    }
    int m=(l+r)>>1;
    return query(root<<1,l,m,ql,qr)+query(root<<1|1,m+1,r,ql,qr);
}
int n,m,op,x,y,k;
signed main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)

        cin>>a[i];

}
